def load_images (imagePath):
    data = []
    labels = []
    
    # Put files into lists 
    image_files = sorted([os.path.join(imagePath, file) for file in os.listdir(imagePath)])
        
    print("path {} images {}".format(imagePath, len(image_files)))

    # pre process images
    for img_name in image_files:
        # Load file and resize to be a fixed IMG_SIZE pixels, ignoring aspect ratio
        image = cv2.imread(img_name, cv2.IMREAD_GRAYSCALE) # gray
        image = cv2.resize(image, (IMG_SIZE, IMG_SIZE))
        # update the data and labels lists, respectively
        data.append(image)
        
        l = img_name.split("/")[-1]
        # labels.append(re.search("[a-zA-Z]*", l.split("_")[0]).group(0))
        labels.append(l.split("_")[0])       
            
    # convert the data into a NumPy array, then preprocess it by scaling
    # all pixel intensities to the range [0, 1]
    data_ = np.array(data, dtype="float") / 255.0
              
    data_ = np.array([i for i in data_]).reshape(-1,IMG_SIZE,IMG_SIZE,1)    
        
    return data_, labels

def get_model( kernel_size=5, drp_out=0.4, fc_size=[64, 128], LRate = 0.0001):
    
    tf.compat.v1.reset_default_graph()
    model_initial = Sequential()

    model_initial.add(Conv2D(32, (kernel_size,kernel_size),activation ="relu", input_shape=(IMG_SIZE,IMG_SIZE,1)))  # 3 for RGB, 1 for gray
    model_initial.add(BatchNormalization())
    model_initial.add(MaxPooling2D((2,2), strides=2))

    model_initial.add(Conv2D(32, (kernel_size,kernel_size),activation ="relu"))  # 3 for RGB, 1 for gray
    model_initial.add(BatchNormalization())
    model_initial.add(MaxPooling2D((2,2), strides=2))
    
    model_initial.add(Conv2D(64, (kernel_size,kernel_size),activation ="relu"))
    model_initial.add(BatchNormalization())
    model_initial.add(MaxPooling2D((2,2), strides=2))
    
    model_initial.add(Flatten())
    
    # Dense layer whose output size is fixed in the hyper parameter: fc_size=32 
    # add few dense layers
    for fc in fc_size:
        model_initial.add(Dense(fc,activation="relu"))
        
    # Dropout layer with dropout probability 0.4 
    model_initial.add(Dropout(drp_out))

    #Predict the class by doing a softmax 
    model_initial.add(Dense(350, activation="softmax"))

    opt = Adam(learning_rate=LRate) 

    model_initial.compile(loss="categorical_crossentropy", optimizer=opt, metrics=["accuracy"])
    
    
    return model_initial
